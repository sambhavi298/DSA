import java.util.HashMap;
class Solution {
    public int[] twoSum(int[] nums, int target) {
        HashMap<Integer, Integer> seen = new HashMap<>();
        for (int i = 0; i < nums.length; i++){
            int x = nums[i];
            int y = target - x;

            if (seen.containsKey(y)){
                return new int[] { seen.get(y), i};
            }
            seen.put(x,i);
        }

        return null;
    }
        
}
